open System

let result =
    let isPalindrome i = 
        let s = i |> string
        s |> Seq.rev |> String.Concat = s 
    seq[ 100 .. 999 ]
    |> Seq.collect (fun i -> seq[ 100 .. 999 ] |> Seq.map (fun j -> i * j))
    |> Seq.sortDescending
    |> Seq.find (fun  x -> isPalindrome x)
